            function setupPagination(tabId, itemsPerPage) {
                const grid = document.querySelector(`#${tabId} .general-section .card-grid`);
                if (!grid) return;
                const cards = Array.from(grid.querySelectorAll('.card'));
                const totalPages = Math.max(1, Math.ceil(cards.length / itemsPerPage));
                let currentPage = 1;
                const pagination = document.getElementById(`${tabId}-pagination`);
                if (!pagination) return;
                function renderControls() {
                    const parts = [];
                    parts.push(`<button class="page-btn prev" ${currentPage === 1 ? 'disabled' : ''} data-page="${currentPage - 1}">ì´ì „</button>`);
                    for (let i = 1; i <= totalPages; i++) {
                        parts.push(`<button class="page-btn ${i === currentPage ? 'active' : ''}" data-page="${i}">${i}</button>`);
                    }
                    parts.push(`<button class="page-btn next" ${currentPage === totalPages ? 'disabled' : ''} data-page="${currentPage + 1}">ë‹¤ìŒ</button>`);
                    pagination.innerHTML = parts.join('');
                }
                function renderPage(page) {
                    currentPage = page;
                    cards.forEach((card, idx) => {
                        card.style.display = (Math.floor(idx / itemsPerPage) + 1) === page ? '' : 'none';
                    });
                    renderControls();
                }
                pagination.addEventListener('click', (e) => {
                    const btn = e.target.closest('.page-btn');
                    if (!btn) return;
                    const page = parseInt(btn.dataset.page, 10);
                    if (!Number.isFinite(page)) return;
                    if (page < 1 || page > totalPages) return;
                    renderPage(page);
                });
                renderPage(1);
            }
