        function generateCalendar(year, month) {
            const firstDay = new Date(year, month, 1);
            const lastDay = new Date(year, month + 1, 0);
            const prevLastDay = new Date(year, month, 0);
            const firstDayOfWeek = firstDay.getDay();
            const lastDateOfMonth = lastDay.getDate();
            const prevLastDate = prevLastDay.getDate();
            
            const calendarGrid = document.getElementById('calendarGrid');
            calendarGrid.innerHTML = '';
            
            // ÏöîÏùº Ìó§Îçî
            const weekDays = ['Ïùº', 'Ïõî', 'Ìôî', 'Ïàò', 'Î™©', 'Í∏à', 'ÌÜ†'];
            weekDays.forEach(day => {
                const dayHeader = document.createElement('div');
                dayHeader.className = 'calendar-day-header';
                dayHeader.textContent = day;
                calendarGrid.appendChild(dayHeader);
            });
            
            // Ïù¥Ï†Ñ Îã¨ ÎÇ†Ïßú
            for (let i = firstDayOfWeek - 1; i >= 0; i--) {
                const dayElement = createDayElement(prevLastDate - i, true, false, false);
                calendarGrid.appendChild(dayElement);
            }
            
            // ÌòÑÏû¨ Îã¨ ÎÇ†Ïßú
            for (let date = 1; date <= lastDateOfMonth; date++) {
                const dateStr = `${year}-${String(month + 1).padStart(2, '0')}-${String(date).padStart(2, '0')}`;
                const hasBanner = bannerData[dateStr] && bannerData[dateStr].length > 0;
                const isToday = isDateToday(year, month, date);
                const isSelected = isDateSelected(year, month, date);
                
                const dayElement = createDayElement(date, false, hasBanner, isToday, isSelected);
                dayElement.dataset.date = dateStr;
                calendarGrid.appendChild(dayElement);
            }
            
            // Îã§Ïùå Îã¨ ÎÇ†Ïßú
            const remainingDays = 42 - (firstDayOfWeek + lastDateOfMonth);
            for (let date = 1; date <= remainingDays; date++) {
                const dayElement = createDayElement(date, true, false, false);
                calendarGrid.appendChild(dayElement);
            }
            
            // Ï∫òÎ¶∞Îçî Ï†úÎ™© ÏóÖÎç∞Ïù¥Ìä∏
            document.getElementById('calendarTitle').textContent = `${year}ÎÖÑ ${month + 1}Ïõî`;
        }
        
        function createDayElement(date, isOtherMonth, hasBanner, isToday, isSelected) {

            if (hasBanner) dayElement.classList.add('has-banner');

            if (isSelected) dayElement.classList.add('selected');

                indicator.className = 'banner-indicator';
                dayElement.appendChild(indicator);
            }
            
            dayElement.addEventListener('click', function() {
                if (!isOtherMonth) {
                    selectDate(this.dataset.date);
                }
            });
            
            return dayElement;
        }
        
        function isDateToday(year, month, date) {
            const today = new Date();
            return year === today.getFullYear() && 
                   month === today.getMonth() && 
                   date === today.getDate();
        }
        
        function isDateSelected(year, month, date) {
            return year === selectedDate.getFullYear() &&
